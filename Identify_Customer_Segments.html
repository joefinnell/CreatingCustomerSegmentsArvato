# Function
def convert_missing_values(data=azdias, feat_info=feat_info):
    for column_name in data.columns:
        # Grab the missing or unkown values for the attribute name and convert them to list
        missing_values_raw = feat_info[feat_info['attribute']==column_name]["missing_or_unknown"].iloc[0]
        missing_values = missing_values_raw.replace('[','')
        missing_values = missing_values.replace(']','')
        missing_values_list = missing_values.split(',')
        # If blank list there are no mapping codes
        if missing_values_list == ['']:
            continue
        # Go through the and replace the missing values
        for value in missing_values_list:
            if data[column_name].dtype == 'object':
                data.loc[data[column_name]==value,column_name]=np.nan 
            else:   
                data.loc[data[column_name]==int(value),column_name]=np.nan 
    return data

The further a weight is from zero, the more the principal component is in the direction of the corresponding feature. If two features have large weights of the same sign (both positive or both negative), then increases in one tend expect to be associated with increases in the other. def run_kmeans(centers, data = X_pca_26):
    scores = []
    centers = range(1,centers+1)
    for center in centers:
        print(f'Training KMeans with {center} Centers')
        k_means_i = KMeans(center)
        model_i = k_means_i.fit(data)
        score = np.abs(model_i.score(data))
        scores.append(score)
